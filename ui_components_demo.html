Dieses Beispiel zeigt, wie Dashboard-Komponenten und Schema-UI-Komponenten gemeinsam verwendet werden können.

Code-Beispiele

BentoGrid mit AdvancedFilter

import { BentoGrid, AdvancedFilter, lightweightAdapter } from "./src/ui_components";

const grid = new BentoGrid({
  elementId: "grid-container",
  items: myData,
  columns: {
    name: { label: "Name", sortable: true },
    status: { label: "Status", sortable: true },
  },
  useAdvancedFilter: true,
  advancedFilterId: "filter-container",
  adapter: lightweightAdapter
});

DynamicMetricTile

import { DynamicMetricTile, lightweightAdapter } from "./src/ui_components";

new DynamicMetricTile({
  elementId: "metric-container",
  title: "Aktive Benutzer",
  value: 1234,
  previousValue: 1100,
  format: "number",
  icon: "bi-person",
  adapter: lightweightAdapter
});

Schema UI mit unifiedAdapter

import { EnhancedProfileForm, unifiedAdapter } from "./src/ui_components";

const profileForm = new EnhancedProfileForm({
  elementId: "profile-form",
  adapter: unifiedAdapter,
  onSave: (data) => {
    console.log("Profil gespeichert:", data);
  }
});
