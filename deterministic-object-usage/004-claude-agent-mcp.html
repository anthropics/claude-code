2. Agent Interaction Sequence — From Prompt to MCP Tool Call

%%{init: {'theme': 'dark', 'themeVariables': {'actorBkg': '#58a6ff', 'actorTextColor': '#0d1117', 'actorLineColor': '#58a6ff', 'signalColor': '#c9d1d9', 'signalTextColor': '#c9d1d9', 'noteBkgColor': '#21262d', 'noteTextColor': '#c9d1d9'}}}%%
sequenceDiagram
    participant U as User
    participant M as Main Agent
    participant HR as Hook Runner
    participant TR as Tool Router
    participant MCP as MCP Client
    participant S as MCP Server
    participant SA as Subagent

    U->>HR: prompt text
    HR->>HR: UserPromptSubmit hook
    HR->>M: validated prompt

    M->>TR: tool_use request
    TR->>HR: PreToolUse hook
    HR-->>TR: allow / deny / modify

    alt MCP Tool
        TR->>MCP: call mcp__server__tool
        MCP->>S: JSON-RPC request
        S-->>MCP: tool result
        MCP-->>TR: result
    else Task (Subagent)
        TR->>SA: spawn subagent
        SA->>SA: autonomous execution
        SA-->>TR: result
        TR->>HR: SubagentStop hook
    else Built-in Tool
        TR->>TR: execute locally
    end

    TR->>HR: PostToolUse hook
    HR-->>TR: proceed
    TR-->>M: tool_result
    M-->>U: response

5. Object Relationship — MCP Server, Tools, and Params

%%{init: {'theme': 'dark', 'themeVariables': {'primaryColor': '#79c0ff', 'primaryTextColor': '#c9d1d9', 'primaryBorderColor': '#30363d', 'lineColor': '#79c0ff'}}}%%
classDiagram
    class MCPServer {
        +string name
        +string transport [stdio|sse|http]
        +string scope [user|project]
        +string[] env_vars
        +string instructions
        +string headersHelper
        +connect()
        +disconnect()
        +listTools()
        +listResources()
        +callTool()
    }

    class MCPTool {
        +string name
        +string description
        +object inputSchema
        +object outputSchema
        +object annotations
        +boolean enabled
        +execute(params)
    }

    class MCPResource {
        +string uri
        +string name
        +string mimeType
        +read()
    }

    class MCPTransport {
        <>
        +send(message)
        +receive() message
    }

    class StdioTransport {
        +string command
        +string[] args
        +number timeout
    }

    class SSETransport {
        +string url
        +object headers
        +string oauth_config
    }

    class HTTPTransport {
        +string url
        +object headers
        +string oauth_config
    }

    class SubagentConfig {
        +string name
        +string model
        +string[] tools
        +string permissionMode
        +string systemPrompt
        +boolean recursive
    }

    class HookConfig {
        +string event
        +string command
        +number timeout
        +string[] matchers
    }

    class SkillConfig {
        +string name
        +string description
        +string agent
        +string model
        +string content
        +string[] triggers
    }

    MCPServer "1" --> "*" MCPTool : exposes
    MCPServer "1" --> "*" MCPResource : provides
    MCPServer "1" --> "1" MCPTransport : uses
    MCPTransport <|-- StdioTransport
    MCPTransport <|-- SSETransport
    MCPTransport <|-- HTTPTransport
    SubagentConfig --> MCPServer : may access
    HookConfig --> MCPTool : intercepts
    SkillConfig --> SubagentConfig : configures
